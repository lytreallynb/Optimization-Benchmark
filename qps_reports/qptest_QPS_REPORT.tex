\begin{align}
\min\quad f(x) &= 0 + 1.5\,c_{1} - 2\,c_{2} \nonumber\\
&\quad + \frac{1}{2} \sum_{i,j} Q_{ij} x_i x_j\label{eq:objective}
\end{align}

\textbf{二次项矩阵特征:}
\begin{itemize}
\item 对角线项: 2 个
\item 非对角线项: 1 个
\item 矩阵类型: 对称正定
\end{itemize}

\textbf{约束条件:}
\begin{align}
2\,c_{1} + 1\,c_{2} &\geq 2 \nonumber
\end{align}

\begin{align}
-1\,c_{1} + 2\,c_{2} &\leq 6 \nonumber
\end{align}

\textbf{变量边界:}
\begin{itemize}
